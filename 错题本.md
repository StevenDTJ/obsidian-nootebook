## 🔀 排序方式

> 点击按钮切换排序方式（当前：<span id="sort-display">按序号</span>）

```dataviewjs
// 创建按钮
const container = dv.container;
container.innerHTML = '';

const btnContainer = container.createEl('div', { cls: 'sort-buttons' });

const sortById = btnContainer.createEl('button', { text: '📋 按序号排序' });
const sortByDate = btnContainer.createEl('button', { text: '📅 按日期排序' });

sortById.style.marginRight = '10px';
sortById.style.padding = '5px 15px';
sortByDate.style.padding = '5px 15px';

// 获取当前排序方式
let sortMode = localStorage.getItem('cuoti-sort-mode') || 'id';

function updateDisplay() {
    const pages = dv.pages('"错题"');
    let sortedPages;

    if (sortMode === 'date') {
        sortedPages = pages.sort(p => p.date, 'desc');
        document.getElementById('sort-display').textContent = '按日期（新→旧）';
    } else {
        sortedPages = pages.sort(p => parseInt(p.file.name), 'asc');
        document.getElementById('sort-display').textContent = '按序号（1→N）';
    }

    // 显示表格
    const tableData = sortedPages.map(p => [
        p.date ? dv.date(p.date).toFormat('yyyy-MM-dd') : '-',
        p.题型 || '-',
        p.模块 || '-',
        p.知识点 || '-',
        p.正确答案 || '-',
        p.错误原因 || '-'
    ]);

    dv.table(['日期', '题型', '模块', '知识点', '正确答案', '错误原因'], tableData);
}

sortById.onclick = () => {
    sortMode = 'id';
    localStorage.setItem('cuoti-sort-mode', 'id');
    updateDisplay();
};

sortByDate.onclick = () => {
    sortMode = 'date';
    localStorage.setItem('cuoti-sort-mode', 'date');
    updateDisplay();
};

updateDisplay();
```
