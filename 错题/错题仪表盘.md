## 二、各模块错题量

```dataviewjs
const pages = dv.pages('"错题"').where(p => p.模块);
const grouped = {};
pages.forEach(p => {
    const key = p.模块;
    grouped[key] = (grouped[key] || 0) + 1;
});
const data = Object.entries(grouped).map(([模块, 数量]) => ({模块, 数量}));
data.sort((a, b) => b.数量 - a.数量);
dv.table(["模块", "错题数量"], data.map(d => [d.模块, d.数量]));
```

## 四、高频错因分析

```dataviewjs
const pages = dv.pages('"错题"').where(p => p.错误原因 && p.错误原因 !== "略");
const grouped = {};
pages.forEach(p => {
    const key = p.错误原因;
    if (!grouped[key]) {
        grouped[key] = { count: 0, files: [] };
    }
    grouped[key].count += 1;
    // 提取题号（文件名）
    const fileName = p.file.name;
    grouped[key].files.push(`[[错题/${fileName}|${fileName}]]`);
});
const data = Object.entries(grouped).map(([错误原因, info]) => ({错误原因, 出现次数: info.count, 错误题号: info.files.join(", ")}));
data.sort((a, b) => b.出现次数 - a.出现次数);
dv.table(["错误原因", "出现次数", "错误题号"], data.map(d => [d.错误原因, d.出现次数, d.错误题号]));
```

## 六、近期错题（最近7天）

```dataview
TABLE WITHOUT ID
  date(date) as 日期,
  题型,
  模块,
  知识点,
  正确答案,
  错误原因
FROM "错题"
WHERE date >= date(today) - dur(7 days)
SORT date DESC
```
